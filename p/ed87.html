Node-初探koa

const koa = require('koa')
//注意引入方式
// 方式一
const router = require('koa-router')() 
// 方式二
// const Router = require('koa-router')
// const router = new Router()

const render = require('koa-art-template')
const path = require('path')
const bodyParser = require('koa-bodyparser')
const static = require('koa-static')
const session = require('koa-session')

const app = new koa()

// 路由级中间件
router.get('/test1',async (ctx, next) => {
    console.log('mmmm')
     //koa中cookies无法直接设置中文
    // ctx.cookies.set('cookieTest', '中文')  
    var cookieTest = new Buffer.from('中文').toString('base64')
    ctx.cookies.set('cookieTest', cookieTest)  
    ctx.session.sessionTest = "会话" //没有中文问题
    await next()
})

router.get('/test1',async (ctx) => {
    let content = "333"
    console.log(ctx.query)
    console.log(ctx.state.userinfo)

    await ctx.render('index',{
        content,
    })
})
// 动态路由
router.get('/test2/:id',async (ctx) => {
    console.log(ctx.params)
    var data1 = ctx.cookies.get('cookieTest')
    var data2 = new Buffer.from(data1, 'base64').toString()
    console.log(data2)
    console.log(ctx.session.sessionTest) 
})

// 登录，post请求
router.get('/test3', async (ctx) => {
    await ctx.render('login')
})
router.post('/login', async (ctx) => {
    console.log(ctx.request.body)
    ctx.body = ctx.request.body
})


// 配置koa-art-template
render(app, {
    root: path.join(__dirname, 'views'),
    extname: '.html',
    debug: process.env.NODE_ENV !== 'production'
})
// 配置session的中间件
app.keys = ['some secret hurr'];

const CONFIG = {
  key: 'koa:sess', /** (string) cookie key (default is koa:sess) */
  /** (number || 'session') maxAge in ms (default is 1 days) */
  /** 'session' will result in a cookie that expires when session/browser is closed */
  /** Warning: If a session cookie is stolen, this cookie will never expire */
  maxAge: 86400000, /** 需要修改 */
  autoCommit: true, /** (boolean) automatically commit headers (default true) */
  overwrite: true, /** (boolean) can overwrite or not (default true) */
  httpOnly: true, /** (boolean) 只有服务端能访问 (default true) */
  signed: true, /** (boolean) signed or not (default true) */
  rolling: false, /** 重新发请求，重新设置session  (default is false) */
  renew: true, /** 需要修改false->true 用户有操作，重新设置session (default is false)*/
  sameSite: null, /** (string) session cookie sameSite options (default null, don't set it) */
};

app.use(session(CONFIG, app));


app
    .use(static('./static'))    //配置静态web服务的中间件
    .use(bodyParser())      //获取表单数据中间件,要写在路由前面，否则不生效
    // 写中间件配置公共的信息，要写在路由前面，否则不生效
    .use(async (ctx, next) => {           
        ctx.state.userinfo = 'ed'
        await next()          
    })
    .use(router.routes())       // 启动路由
    .use(router.allowedMethods())   //推荐写上此配置

    .use(async (ctx, next) => {     //应用级中间件
        console.log(new Date())     //匹配路由之前打印日期
        await next()    //當前路由匹配完成以后继续向后匹配

        if(ctx.status === 404) {    //错误处理中间件
            ctx.status = 404
            ctx.body = '404了'
        }
    })


app.listen(3000, () => {
    console.log(3000)
})
