Vue-vue-element-admin登录思路与代码

1. 流程思路

(1)用户登录
  ↓
  - 成功
    ↓
    - 把token保存至vuex
    - 把token保存至Cookies
    - 设置vuex中的用户信息
    - 设置用户路由
    - 用addRutes动态添加路由并跳转至首页
  ↓
  - 失败
    ↓
    - 继续登录

(2)拦截路由变化
  ↓
  - 判断vuex中是否有token
    ↓
    - 有token
      ↓
      - 判断是否跳转路径为登录页
        ↓
        - 路径为登录页
        - 跳转至首页
        ↓
        - 路径不为登录页
          - 验证用户信息
            ↓
            - 用户信息通过
             - 跳转至首页
            ↓
            - 用户信息不通过
             - 设置vuex中的用户信息
             - 设置用户路由
             - 用addRutes动态添加路由并跳转至首页
    ↓
    - 没有token
      ↓
      - 跳转登录页面

2. 详细代码分析

（1）点击提交

登录页中的点击提交触发事件

具体代码：@/views/login/index.vue

import { validUsername } from '@/utils/validate'

handleLogin() {
    // validate 是做一个登录的前端校验，具体做了一个验证名字
    this.$refs.loginForm.validate(valid => {
        if (valid) {
            this.loading = true
            // 触发vuex中user模块中，action的login事件
            this.$store.dispatch('user/login', this.loginForm)
                .then(() => {
                //成功则跳转
                this.$router.push({ path: this.redirect || '/', query: this.otherQuery })
                this.loading = false
            })
                .catch(() => {
                this.loading = false
            })
        } else {   //前端校验，失败提示，不会发起请求
            this.$notify.error({
                title: '失败！',
                message: '请检查您的输入是否规范!',
                duration: 2000
            })
            return false
        }
    })
  },

Ps: 校验名字，具体也是要根据自己的系统来修改，具体代码：@/utils/validate.js

// 简单写了不能为空
export function validUsername(str) {
  return str.trim().length > 0
}

Ps: 关于token工具类，具体代码：@/utils/auth.js


import Cookies from 'js-cookie'

const TokenKey = 'Admin-Token'

//1.将token存入cookies， 2.在cookies取出token,3.把cookies中的token移除
export function getToken() {
  return Cookies.get(TokenKey)
}

export function setToken(token) {
  return Cookies.set(TokenKey, token)
}

export function removeToken() {
  return Cookies.remove(TokenKey)
}

（3）设置用户信息

触发vuex的action中的getInfo事件

  1. 什么时候触发vuex的action中的getInfo事件？

触发该事件，是写在拦截路由的文件中，

具体代码：@/permission.js

该文件代码的思路可参考上文提到的 流程思路中的拦截路由变化

const { roles } = await store.dispatch('user/getInfo')
const accessRoutes = await store.dispatch('permission/generateRoutes', roles)
router.addRoutes(accessRoutes)

这三行代码分别对应

- 设置vuex中的用户信息
- 设置用户路由
- 用addRutes动态添加路由并跳转至首页
  1. getInfo事件具体代码：@/store/modules/user.js
// getInfo接口是通过token,获取用户信息（如用户权限，用户名，用户头像等）
import { login, getInfo } from '@/api/Login/user'

const state = {
  name: '',
  avatar: '',
  introduction: '',
  roles: []
}

const mutations = {
  SET_INTRODUCTION: (state, introduction) => {
    state.introduction = introduction
  },
  SET_NAME: (state, name) => {
    state.name = name
  },
  SET_AVATAR: (state, avatar) => {
    state.avatar = avatar
  },
  SET_ROLES: (state, roles) => {
    state.roles = roles
  }
    .....
}

const actions = {
  // get user info
  getInfo({ commit, state }) {
    return new Promise((resolve, reject) => {
      getInfo(state.token).then(response => {
        const { data } = response
        if (!data) {
          reject('Verification failed, please Login again.')
        }
        const { roles, name, avatar, introduction } = data
        // roles must be a non-empty array
        if (!roles || roles.length <= 0) {
          reject('getInfo: roles must be a non-null array!')
        }
        commit('SET_ROLES', roles)
        commit('SET_NAME', name)
        commit('SET_AVATAR', avatar)
        commit('SET_INTRODUCTION', introduction)
        resolve(data)
      }).catch(error => {
        reject(error)
      })
    })
  },
    ......
  }
}

export default {
  namespaced: true,
  state,
  mutations,
  actions
}

（4）设置路由

触发vuex的action中的generateRoutes事件

触发的时机也已在第三点中提到

具体代码：@/store/modules/permission.js

import { asyncRoutes, constantRoutes } from '@/router'

// 判断是否有权限，如果没有设置roles，则此页默认没有权限，像以下这种就是设置权限为admin
/* meta: {
          title: 'Role Permission',
          roles: ['admin']
  }
*/ 
function hasPermission(roles, route) {
  if (route.meta && route.meta.roles) {
    return roles.some(role => route.meta.roles.includes(role))
  } else {
    return true
  }
}

// 过滤掉权限不一致的路由，采用递归的方式，即可探索子节点是否权限不一致
export function filterAsyncRoutes(routes, roles) {
  const res = []

  routes.forEach(route => {
    const tmp = { ...route }
    if (hasPermission(roles, tmp)) {
      if (tmp.children) {
        tmp.children = filterAsyncRoutes(tmp.children, roles)
      }
      res.push(tmp)
    }
  })

  return res
}

const state = {
  routes: [],
  addRoutes: []
}

const mutations = {
  SET_ROUTES: (state, routes) => {
    state.addRoutes = routes
    state.routes = constantRoutes.concat(routes)
  }
}


const actions = {
  generateRoutes({ commit }, roles) {
    return new Promise(resolve => {
      let accessedRoutes
      // 如果是admin，则不过滤，直接将动态路由都添加，不是admin则过滤
      if (roles.includes('admin')) {
        accessedRoutes = asyncRoutes || []
      } else {
        accessedRoutes = filterAsyncRoutes(asyncRoutes, roles)
      }
      commit('SET_ROUTES', accessedRoutes)
      resolve(accessedRoutes)
    })
  }
}

export default {
  namespaced: true,
  state,
  mutations,
  actions
}
